char gradeIt(int mark, int maxMark)
{
	int totalMark = (static_cast<double>(mark) / static_cast<double>(maxMark)) * 100;
	if (totalMark <= 50)
	{
		return 'F';
	}
	if (totalMark > 50 && totalMark <= 59)
	{
		return 'D';
	}
	else if (totalMark > 59 && totalMark <= 69)
	{
		return 'C';
	}
	else if (totalMark > 69 && totalMark <= 79)
	{
		return 'B';
	}
	else
	{
		return 'A';
	}

}
